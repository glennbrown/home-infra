  - service_name: influxdb
    active: false
    image: influxdb:1.8.10-alpine
    container_name: influxdb
    include_global_env_vars: false

      - "TZ={{ ntp_timezone }}"

      - '"traefik.http.routers.influxdb.tls=true"'
      - '"traefik.http.routers.influxdb.tls.certresolver=cloudflare"'
      - '"traefik.http.routers.influxdb.tls.domains[0].main={{ local_domain }}"'
      - '"traefik.http.routers.influxdb.tls.domains[0].sans=*.{{ local_domain }},*.{{ ansible_hostname[0] }}.{{ local_domain }}"'
    restart: unless-stopped
